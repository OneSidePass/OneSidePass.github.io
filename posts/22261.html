def f_inspect(e1_get,fps_get,image_name_get):  #检验输入规则mod
    global count
    e1_get = e1_get
    fps_get =fps_get
    image_name_get = image_name_get
    path = os.getcwd()
    os.system(path[0] + path[1])
    os.system("cd " + path)
    count += 1
    pattern = re.compile('\.+')
    if not pattern.search(e1_get) or not pattern.search(image_name_get):
        count -= 5
    if not fps_get.isdigit() or e1_get == "" or image_name_get == "":
        count -= 5
    if not os.path.exists(f"{path}"+"\\\\output"):
        os.mkdir(f"{path}"+"\\\\output")
    if count == 1:
        os.system("ffmpeg -i "+e1_get+" -r "+fps_get+" -f image2 "+f"{path}"+"\\\\output\\\\"+image_name_get)
        #print("ffmpeg -i "+e1_get+" -r "+fps_get+" -f image2 "+f"{path}"+"\\\\output\\\\"+image_name_get)
    time_c()

def f_inspect4(now_file_e):

def time_c():  #计数器
    global news,count
    if count == 1:
        news.set("开始执行")
    elif count == 0:
        news.set("111")

    elif count < 0:
        news.set("格式错误")
        count = 0
    else:
        news.set("已经执行了\n不要再按了!!!")

def en_cod():
    #规定输入规则
    path_pattern = re.compile(r'^([a-zA-Z]:)(\\\\\w+)*$')
    fps_pat = re.compile(r'\d+')
    file_pat = re.compile(r'^(\w+)\.(\w+)')

    print("ffmpeg将视频分割为N张图片".center(50,"-"))
    path = input("请输入视频路经(例如D:\\\\filedir)(注意：不包括文件名)：")
    while not path_pattern.search(path):
        print("格式错误，请重新输入")
        path = input("请输入视频路经(例如D:\\\\filedir)(注意：不包括文件名)：")
    file = input("请输入文件名(需要加后缀名)（如：out.mp4）:").strip()
    fps = input("输出帧数(常用：25/30/60)：")
    while not fps_pat.search(fps):

    image_name = input("图片命名规则（如：name04%d）(需要加后缀名.jpg或.png)：")
    # print("cd "+path[0]+path[1])

    os.system("cd "+path)
    if not os.path.exists(f"{path}"+"\\\\output"):
        os.mkdir(f"{path}"+"\\\\output")
