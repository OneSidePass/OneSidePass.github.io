  • 公司员工抽奖小程序

import random
import string
staffs = []                       #空列表用于存储员工
for i in range(1,301):            #1-300 （for循环）
    staffs.append(i)              #将1-300分别存入空列表
c = random.sample(staffs,30)      #c为三等奖随机抽取的变量
print(f"三等奖获奖的员工分别为：{c}")
if c in staffs:
    staffs.remove(c)              #remove 删除了三等奖的员工，剩下的是未获奖的员工
b = random.sample(staffs,6)
print(f'二等奖获奖的员工分别为：{b}')
if b in staffs:
    staffs.remove(b)
a = random.sample(staffs,3)
print(f'一等奖获奖的员工分别为：{a}')

import string
import random
count = 0              #计数器
car_nums = []          #空集合存数据
while count < 4:
    for i in range(20):
        n1 = random.choice(string.ascii_uppercase)    #随机选取里面的任意一个对象
        n2 = "".join(random.sample(string.ascii_uppercase + string.digits, 5))
        c_num = (f"京{n1}-{n2}")
        car_nums.append(c_num)
        print(c_num,end=" ")
        if i == 4 or i == 9 or i == 14  :    #换行模块
            print()                          #换行模块
    print()                                  #换行模块
    choice = input("请输入你选择的车牌,如不满意请输入no:").strip() #strip 方法去掉头尾空格及特殊字符
    if choice in car_nums:
        print(f"恭喜您喜提新车牌：{choice}")
        exit()
    else:
        if count < 3:
            print()
            print("-------------------抱歉请重试-------------------")
            print()
    count += count + 1                         #计数器
choice2 = []
while choice2 not in car_nums:                 #死循环
    choice2 = input("三次选择结束请选择车牌：").strip()
print(f"恭喜您喜提新车牌： {choice2}！！！")

name = {                              #字典
    "osp" : [22,'student',1500],
    "佩奇" : [14,"student",0],
    "umr"  : "好吃懒做"
}
#取值

print('字典  ： ',name)
print('取值（依据key取出value） ：     ',name["umr"])

#增加
name["tsf"] = "活泼可爱"
print('增加 ：  ',name)

#修改
name["osp"] = "帅气动人"
name["佩奇"] = "古灵精怪"
print('修改 ：  ',name)

#删除
del name["佩奇"]
name.pop("tsf")
print('删除 ：  ',name)

#清空
name.clear()
print('清空 ：  ',name)
